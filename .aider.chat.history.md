from openai import OpenAI
from typing import List, Dict

from src.config import Config

class NewsSummarizerAgent(BaseAgent):

class NewsFetcherAgent(BaseAgent):

from openai import OpenAI
from typing import List, Dict
from agents.base import BaseAgent
from config import Config

class NewsSummarizerAgent(BaseAgent):
    """Agent responsible for summarizing news articles using an LLM."""
    def __init__(self):
        super().__init__("NewsSummarizer")
        self.client = OpenAI(api_key=Config.OPENAI_API_KEY)

    def run(self, articles: List[Dict]) -> List[Dict]:
        summarized = []
        for article in articles:
            prompt = (
                f"Summarize the following news article in 2-3 concise sentences. "
                f"Focus on the key facts and implications.\n\n"
                f"Title: {article['title']}\n"
                f"Content: {article['summary']}"
            )
            try:
                response = self.client.chat.completions.create(
                    model=Config.LLM_MODEL,
                    messages=[{"role": "user", "content": prompt}],
                    max_tokens=150,
                    temperature=0.3
                )
                summary_text = response.choices[0].message.content.strip()
                summarized.append({**article, "ai_summary": summary_text})
                print(f"[{self.name}] Summarized: {article['title']}")
            except Exception as e:
                print(f"[{self.name}] Error summarizing '{article['title']}': {e}")
                summarized.append({**article, "ai_summary": "Failed to generate summary."})
        return summarized

class NewsFetcherAgent(BaseAgent):

        self.exclude_keywords = ["sports", "entertainment", "lifestyle", "celebrity", "opinion", "editorial"]

    async def run(self) -> List[Dict]:
        articles = []
        async with aiohttp.ClientSession() as session:
            for url in settings.RSS_FEEDS:
                try:
                    async with session.get(url, timeout=10) as resp:
                        if resp.status == 200:
                            xml = await resp.text()
                            feed = feedparser.parse(xml)
                            for entry in feed.entries:
                                if self._is_relevant(entry):
                                    articles.append({
                                        "title": entry.get("title", ""),
                                        "link": entry.get("link", ""),
                                        "summary": entry.get("summary", ""),
                                        "source": feed.feed.get("title", "Unknown")
                                    })

        return articles

class NewsFetcherAgent(BaseAgent):
    def __init__(self):
        super().__init__("NewsFetcher")
        self.exclude_keywords = ["sports", "entertainment", "lifestyle", "celebrity", "opinion", "editorial", "gossip"]

    async def run(self) -> List[Dict]:
        articles = []
        async with aiohttp.ClientSession() as session:
            for url in settings.RSS_FEEDS:
                try:
                    async with session.get(url, timeout=15) as resp:
                        if resp.status == 200:
                            xml = await resp.text()
                            feed = feedparser.parse(xml)
                            for entry in feed.entries:
                                if self._is_relevant(entry):
                                    articles.append({
                                        "title": entry.get("title", "No Title"),
                                        "link": entry.get("link", ""),
                                        "summary": entry.get("summary", ""),
                                        "source": feed.feed.get("title", "Unknown Source")
                                    })
                except Exception as e:
                    print(f"[{self.name}] Error fetching {url}: {e}")
        print(f"[{self.name}] Fetched {len(articles)} relevant articles.")
        return articles

    def _is_relevant(self, entry: Dict) -> bool:
        text = f"{entry.get('title', '')} {entry.get('summary', '')}".lower()
        return not any(kw in text for kw in self.exclude_keywords)

                status_snapshot TEXT,
                impact_level TEXT,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                FOREIGN KEY(subject_id) REFERENCES subjects(id)

async def get_subject_by_name(name: str):
    async with aiosqlite.connect(settings.DB_PATH) as db:
        db.row_factory = aiosqlite.Row

  - **Dynamic Detection**: LLM automatically identifies and names emerging topics not on the watchlist, adding them to the tracking pool if they meet relevance thresholds.
- **State Management**: Each subject maintains a `latest_status`, `last_updated`, `impact_level`, `relevance_score`, and `development_history`.

## 5. Output & Delivery
- **Platform**: Telegram Bot.
- **Trigger**: Every polling cycle after processing completes.
- **Format**: Structured status report.
  - Unchanged subjects: Display current status concisely.
  - New developments: Highlight with clear markers (e.g., `🔴 NEW DEVELOPMENT:` or `⚡ UPDATE:`).
  - Grouped by impact level or sector for readability.
- **Source Linking & Interactivity**:
  - Every reported item includes a direct MarkdownV2 hyperlink to the original article (`[Title](URL)`).
  - Inline keyboard buttons attached to each message: `🔗 Read Source`, `📖 Full Context`, `📊 Impact Details`.
  - Callback handlers fetch and reply with extended timelines, related articles, or full impact breakdowns without cluttering the main feed.

## 6. Data Storage
